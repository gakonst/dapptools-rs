40/// When running tests, we deploy all external libraries present in the project. To avoid additional

98    /// Deploys the test contract inside the runner from the sending account, and optionally runs
99    /// the `setUp` function on the test contract.
100    pub fn setup(&mut self, call_setup: bool) -> TestSetup {
101        self._setup(call_setup).unwrap_or_else(|err| {
102            if err.to_string().contains("skipped") {
103                TestSetup::skipped(err.to_string())
104            } else {
105                TestSetup::failed(err.to_string())
106            }
107        })
108    }
109
110    fn _setup(&mut self, call_setup: bool) -> Result<TestSetup> {
111        trace!(call_setup, "setting up");
112
113        self.apply_contract_inline_config()?;
114
115        // We max out their balance so that they can deploy and make calls.
116        self.executor.set_balance(self.sender, U256::MAX)?;
117        self.executor.set_balance(CALLER, U256::MAX)?;
118
119        // We set the nonce of the deployer accounts to 1 to get the same addresses as DappTools.
120        self.executor.set_nonce(self.sender, 1)?;
121
122        // Deploy libraries.
123        self.executor.set_balance(LIBRARY_DEPLOYER, U256::MAX)?;
124
125        let mut result = TestSetup::default();
126        for code in self.mcr.libs_to_deploy.iter() {
127            let deploy_result = self.executor.deploy(
128                LIBRARY_DEPLOYER,
129                code.clone(),
130                U256::ZERO,
131                Some(&self.mcr.revert_decoder),
132            );
133
134            // Record deployed library address.
135            if let Ok(deployed) = &deploy_result {
136                result.deployed_libs.push(deployed.address);
137            }
138
139            let (raw, reason) = RawCallResult::from_evm_result(deploy_result.map(Into::into))?;
140            result.extend(raw, TraceKind::Deployment);
141            if reason.is_some() {
142                result.reason = reason;
143                return Ok(result);
144            }
145        }
146
147        let address = self.sender.create(self.executor.get_nonce(self.sender)?);
148        result.address = address;
149
150        // Set the contracts initial balance before deployment, so it is available during
151        // construction
152        self.executor.set_balance(address, self.initial_balance())?;
153
154        // Deploy the test contract
155        let deploy_result = self.executor.deploy(
156            self.sender,
157            self.contract.bytecode.clone(),
158            U256::ZERO,
159            Some(&self.mcr.revert_decoder),
160        );
161
162        result.deployment_failure = deploy_result.is_err();
163
164        if let Ok(dr) = &deploy_result {
165            debug_assert_eq!(dr.address, address);
166        }
167        let (raw, reason) = RawCallResult::from_evm_result(deploy_result.map(Into::into))?;
168        result.extend(raw, TraceKind::Deployment);
169        if reason.is_some() {
170            result.reason = reason;
171            return Ok(result);
172        }
173
174        // Reset `self.sender`s, `CALLER`s and `LIBRARY_DEPLOYER`'s balance to the initial balance.
175        self.executor.set_balance(self.sender, self.initial_balance())?;
176        self.executor.set_balance(CALLER, self.initial_balance())?;
177        self.executor.set_balance(LIBRARY_DEPLOYER, self.initial_balance())?;
178
179        self.executor.deploy_create2_deployer()?;
180
181        // Optionally call the `setUp` function
182        if call_setup {
183            trace!("calling setUp");
184            let res = self.executor.setup(None, address, Some(&self.mcr.revert_decoder));
185            let (raw, reason) = RawCallResult::from_evm_result(res)?;
186            result.extend(raw, TraceKind::Setup);
187            result.reason = reason;
188        }
189
190        result.fuzz_fixtures = self.fuzz_fixtures(address);
191
192        Ok(result)
193    }

199    /// Configures this runner with the inline configuration for the contract.
200    fn apply_contract_inline_config(&mut self) -> Result<()> {
201        if self.inline_config.contains_contract(self.name) {
202            let new_config = Arc::new(self.inline_config(None)?);
203            self.tcfg.to_mut().reconfigure_with(new_config);
204            let prev_tracer = self.executor.inspector_mut().tracer.take();
205            self.tcfg.configure_executor(&mut self.executor);
206            // Don't set tracer here.
207            self.executor.inspector_mut().tracer = prev_tracer;
208        }
209        Ok(())
210    }

220    /// Collect fixtures from test contract.
221    ///
222    /// Fixtures can be defined:
223    /// - as storage arrays in test contract, prefixed with `fixture`
224    /// - as functions prefixed with `fixture` and followed by parameter name to be fuzzed
225    ///
226    /// Storage array fixtures:
227    /// `uint256[] public fixture_amount = [1, 2, 3];`
228    /// define an array of uint256 values to be used for fuzzing `amount` named parameter in scope
229    /// of the current test.
230    ///
231    /// Function fixtures:
232    /// `function fixture_owner() public returns (address[] memory){}`
233    /// returns an array of addresses to be used for fuzzing `owner` named parameter in scope of the
234    /// current test.
235    fn fuzz_fixtures(&mut self, address: Address) -> FuzzFixtures {
236        let mut fixtures = HashMap::default();
237        let fixture_functions = self.contract.abi.functions().filter(|func| func.is_fixture());
238        for func in fixture_functions {
239            if func.inputs.is_empty() {
240                // Read fixtures declared as functions.
241                if let Ok(CallResult { raw: _, decoded_result }) =
242                    self.executor.call(CALLER, address, func, &[], U256::ZERO, None)
243                {
244                    fixtures.insert(fixture_name(func.name.clone()), decoded_result);
245                }
246            } else {
247                // For reading fixtures from storage arrays we collect values by calling the
248                // function with incremented indexes until there's an error.
249                let mut vals = Vec::new();
250                let mut index = 0;
251                loop {
252                    if let Ok(CallResult { raw: _, decoded_result }) = self.executor.call(
253                        CALLER,
254                        address,
255                        func,
256                        &[DynSolValue::Uint(U256::from(index), 256)],
257                        U256::ZERO,
258                        None,
259                    ) {
260                        vals.push(decoded_result);
261                    } else {
262                        // No result returned for this index, we reached the end of storage
263                        // array or the function is not a valid fixture.
264                        break;
265                    }
266                    index += 1;
267                }
268                fixtures.insert(fixture_name(func.name.clone()), DynSolValue::Array(vals));
269            };
270        }
271        FuzzFixtures::new(fixtures)
272    }

274    /// Runs all tests for a contract whose names match the provided regular expression
275    pub fn run_tests(mut self, filter: &dyn TestFilter) -> SuiteResult {
276        let start = Instant::now();
277        let mut warnings = Vec::new();
278
279        // Check if `setUp` function with valid signature declared.
280        let setup_fns: Vec<_> =
281            self.contract.abi.functions().filter(|func| func.name.is_setup()).collect();
282        let call_setup = setup_fns.len() == 1 && setup_fns[0].name == "setUp";
283        // There is a single miss-cased `setUp` function, so we add a warning
284        for &setup_fn in setup_fns.iter() {
285            if setup_fn.name != "setUp" {
286                warnings.push(format!(
287                    "Found invalid setup function \"{}\" did you mean \"setUp()\"?",
288                    setup_fn.signature()
289                ));
290            }
291        }
292
293        // There are multiple setUp function, so we return a single test result for `setUp`
294        if setup_fns.len() > 1 {
295            return SuiteResult::new(
296                start.elapsed(),
297                [("setUp()".to_string(), TestResult::fail("multiple setUp functions".to_string()))]
298                    .into(),
299                warnings,
300            )
301        }
302
303        // Check if `afterInvariant` function with valid signature declared.
304        let after_invariant_fns: Vec<_> =
305            self.contract.abi.functions().filter(|func| func.name.is_after_invariant()).collect();
306        if after_invariant_fns.len() > 1 {
307            // Return a single test result failure if multiple functions declared.
308            return SuiteResult::new(
309                start.elapsed(),
310                [(
311                    "afterInvariant()".to_string(),
312                    TestResult::fail("multiple afterInvariant functions".to_string()),
313                )]
314                .into(),
315                warnings,
316            )
317        }
318        let call_after_invariant = after_invariant_fns.first().is_some_and(|after_invariant_fn| {
319            let match_sig = after_invariant_fn.name == "afterInvariant";
320            if !match_sig {
321                warnings.push(format!(
322                    "Found invalid afterInvariant function \"{}\" did you mean \"afterInvariant()\"?",
323                    after_invariant_fn.signature()
324                ));
325            }
326            match_sig
327        });
328
329        // Invariant testing requires tracing to figure out what contracts were created.
330        // We also want to disable `debug` for setup since we won't be using those traces.
331        let has_invariants = self.contract.abi.functions().any(|func| func.is_invariant_test());
332
333        let prev_tracer = self.executor.inspector_mut().tracer.take();
334        if prev_tracer.is_some() || has_invariants {
335            self.executor.set_tracing(TraceMode::Call);
336        }
337
338        let setup_time = Instant::now();
339        let setup = self.setup(call_setup);
340        debug!("finished setting up in {:?}", setup_time.elapsed());
341
342        self.executor.inspector_mut().tracer = prev_tracer;
343
344        if setup.reason.is_some() {
345            // The setup failed, so we return a single test result for `setUp`
346            let fail_msg = if !setup.deployment_failure {
347                "setUp()".to_string()
348            } else {
349                "constructor()".to_string()
350            };
351            return SuiteResult::new(
352                start.elapsed(),
353                [(fail_msg, TestResult::setup_result(setup))].into(),
354                warnings,
355            )
356        }
357
358        // Filter out functions sequentially since it's very fast and there is no need to do it
359        // in parallel.
360        let find_timer = Instant::now();
361        let functions = self
362            .contract
363            .abi
364            .functions()
365            .filter(|func| is_matching_test(func, filter))
366            .collect::<Vec<_>>();
367        debug!(
368            "Found {} test functions out of {} in {:?}",
369            functions.len(),
370            self.contract.abi.functions().count(),
371            find_timer.elapsed(),
372        );
373
374        let identified_contracts = has_invariants.then(|| {
375            load_contracts(setup.traces.iter().map(|(_, t)| &t.arena), &self.mcr.known_contracts)
376        });
377
378        let test_fail_instances = functions
379            .iter()
380            .filter_map(|func| {
381                TestFunctionKind::classify(&func.name, !func.inputs.is_empty())
382                    .is_any_test_fail()
383                    .then_some(func.name.clone())
384            })
385            .collect::<Vec<_>>();
386
387        if !test_fail_instances.is_empty() {
388            let instances = format!(
389                "Found {} instances: {}",
390                test_fail_instances.len(),
391                test_fail_instances.join(", ")
392            );
393            let fail =  TestResult::fail("`testFail*` has been removed. Consider changing to test_Revert[If|When]_Condition and expecting a revert".to_string());
394            return SuiteResult::new(start.elapsed(), [(instances, fail)].into(), warnings)
395        }
396
397        let test_results = functions
398            .par_iter()
399            .map(|&func| {
400                let start = Instant::now();
401
402                let _guard = self.tokio_handle.enter();
403
404                let _guard;
405                let current_span = tracing::Span::current();
406                if current_span.is_none() || current_span.id() != self.span.id() {
407                    _guard = self.span.enter();
408                }
409
410                let sig = func.signature();
411                let kind = func.test_function_kind();
412
413                let _guard = debug_span!(
414                    "test",
415                    %kind,
416                    name = %if enabled!(tracing::Level::TRACE) { &sig } else { &func.name },
417                )
418                .entered();
419
420                let mut res = FunctionRunner::new(&self, &setup).run(
421                    func,
422                    kind,
423                    call_after_invariant,
424                    identified_contracts.as_ref(),
425                );
426                res.duration = start.elapsed();
427
428                (sig, res)
429            })
430            .collect::<BTreeMap<_, _>>();
431
432        let duration = start.elapsed();
433        SuiteResult::new(duration, test_results, warnings)
434    }

462impl<'a> FunctionRunner<'a> {
463    fn new(cr: &'a ContractRunner<'a>, setup: &'a TestSetup) -> Self {
464        Self {
465            tcfg: match &cr.tcfg {
466                Cow::Borrowed(tcfg) => Cow::Borrowed(tcfg),
467                Cow::Owned(tcfg) => Cow::Owned(tcfg.clone()),
468            },
469            executor: Cow::Borrowed(&cr.executor),
470            cr,
471            address: setup.address,
472            setup,
473            result: TestResult::new(setup),
474        }

481    /// Configures this runner with the inline configuration for the contract.
482    fn apply_function_inline_config(&mut self, func: &Function) -> Result<()> {
483        if self.inline_config.contains_function(self.cr.name, &func.name) {
484            let new_config = Arc::new(self.cr.inline_config(Some(func))?);
485            self.tcfg.to_mut().reconfigure_with(new_config);
486            self.tcfg.configure_executor(self.executor.to_mut());
487        }
488        Ok(())
489    }
490
491    fn run(
492        mut self,
493        func: &Function,
494        kind: TestFunctionKind,
495        call_after_invariant: bool,
496        identified_contracts: Option<&ContractsByAddress>,
497    ) -> TestResult {
498        if let Err(e) = self.apply_function_inline_config(func) {
499            self.result.single_fail(Some(e.to_string()));
500            return self.result;
501        }
502
503        match kind {
504            TestFunctionKind::UnitTest { .. } => self.run_unit_test(func),
505            TestFunctionKind::FuzzTest { .. } => self.run_fuzz_test(func),
506            TestFunctionKind::InvariantTest => {
507                self.run_invariant_test(func, call_after_invariant, identified_contracts.unwrap())
508            }
509            _ => unreachable!(),
510        }
511    }
512
513    /// Runs a single unit test.
514    ///
515    /// Applies before test txes (if any), runs current test and returns the `TestResult`.
516    ///
517    /// Before test txes are applied in order and state modifications committed to the EVM database
518    /// (therefore the unit test call will be made on modified state).
519    /// State modifications of before test txes and unit test function call are discarded after
520    /// test ends, similar to `eth_call`.
521    fn run_unit_test(mut self, func: &Function) -> TestResult {
522        // Prepare unit test execution.
523        if self.prepare_test(func).is_err() {
524            return self.result;
525        }
526
527        // Run current unit test.
528        let (mut raw_call_result, reason) = match self.executor.call(
529            self.sender,
530            self.address,
531            func,
532            &[],
533            U256::ZERO,
534            Some(self.revert_decoder()),
535        ) {
536            Ok(res) => (res.raw, None),
537            Err(EvmError::Execution(err)) => (err.raw, Some(err.reason)),
538            Err(EvmError::Skip(reason)) => {
539                self.result.single_skip(reason);
540                return self.result;
541            }
542            Err(err) => {
543                self.result.single_fail(Some(err.to_string()));
544                return self.result;
545            }
546        };
547
548        let success =
549            self.executor.is_raw_call_mut_success(self.address, &mut raw_call_result, false);
550        self.result.single_result(success, reason, raw_call_result);
551        self.result
552    }
553
554    fn run_invariant_test(
555        mut self,
556        func: &Function,
557        call_after_invariant: bool,
558        identified_contracts: &ContractsByAddress,
559    ) -> TestResult {
560        // First, run the test normally to see if it needs to be skipped.
561        if let Err(EvmError::Skip(reason)) = self.executor.call(
562            self.sender,
563            self.address,
564            func,
565            &[],
566            U256::ZERO,
567            Some(self.revert_decoder()),
568        ) {
569            self.result.invariant_skip(reason);
570            return self.result;
571        };
572
573        let runner = self.invariant_runner();
574        let invariant_config = &self.config.invariant;
575
576        let mut evm = InvariantExecutor::new(
577            self.clone_executor(),
578            runner,
579            invariant_config.clone(),
580            identified_contracts,
581            &self.cr.mcr.known_contracts,
582        );
583        let invariant_contract = InvariantContract {
584            address: self.address,
585            invariant_function: func,
586            call_after_invariant,
587            abi: &self.cr.contract.abi,
588        };
589
590        let failure_dir = invariant_config.clone().failure_dir(self.cr.name);
591        let failure_file = failure_dir.join(&invariant_contract.invariant_function.name);
592        let show_solidity = invariant_config.clone().show_solidity;
593
594        // Try to replay recorded failure if any.
595        if let Ok(mut call_sequence) =
596            foundry_common::fs::read_json_file::<Vec<BaseCounterExample>>(failure_file.as_path())
597        {
598            // Create calls from failed sequence and check if invariant still broken.
599            let txes = call_sequence
600                .iter_mut()
601                .map(|seq| {
602                    seq.show_solidity = show_solidity;
603                    BasicTxDetails {
604                        sender: seq.sender.unwrap_or_default(),
605                        call_details: CallDetails {
606                            target: seq.addr.unwrap_or_default(),
607                            calldata: seq.calldata.clone(),
608                        },
609                    }
610                })
611                .collect::<Vec<BasicTxDetails>>();
612            if let Ok((success, replayed_entirely)) = check_sequence(
613                self.clone_executor(),
614                &txes,
615                (0..min(txes.len(), invariant_config.depth as usize)).collect(),
616                invariant_contract.address,
617                invariant_contract.invariant_function.selector().to_vec().into(),
618                invariant_config.fail_on_revert,
619                invariant_contract.call_after_invariant,
620            ) {
621                if !success {
622                    let _= sh_warn!("\
623                            Replayed invariant failure from {:?} file. \
624                            Run `forge clean` or remove file to ignore failure and to continue invariant test campaign.",
625                        failure_file.as_path()
626                    );
627                    // If sequence still fails then replay error to collect traces and
628                    // exit without executing new runs.
629                    let _ = replay_run(
630                        &invariant_contract,
631                        self.clone_executor(),
632                        &self.cr.mcr.known_contracts,
633                        identified_contracts.clone(),
634                        &mut self.result.logs,
635                        &mut self.result.traces,
636                        &mut self.result.coverage,
637                        &mut self.result.deprecated_cheatcodes,
638                        &txes,
639                        show_solidity,
640                    );
641                    self.result.invariant_replay_fail(
642                        replayed_entirely,
643                        &invariant_contract.invariant_function.name,
644                        call_sequence,
645                    );
646                    return self.result;
647                }
648            }
649        }
650
651        let progress =
652            start_fuzz_progress(self.cr.progress, self.cr.name, &func.name, invariant_config.runs);
653        let invariant_result = match evm.invariant_fuzz(
654            invariant_contract.clone(),
655            &self.setup.fuzz_fixtures,
656            &self.setup.deployed_libs,
657            progress.as_ref(),
658        ) {
659            Ok(x) => x,
660            Err(e) => {
661                self.result.invariant_setup_fail(e);
662                return self.result;
663            }
664        };
665        // Merge coverage collected during invariant run with test setup coverage.
666        self.result.merge_coverages(invariant_result.coverage);
667
668        let mut counterexample = None;
669        let success = invariant_result.error.is_none();
670        let reason = invariant_result.error.as_ref().and_then(|err| err.revert_reason());
671
672        match invariant_result.error {
673            // If invariants were broken, replay the error to collect logs and traces
674            Some(error) => match error {
675                InvariantFuzzError::BrokenInvariant(case_data) |
676                InvariantFuzzError::Revert(case_data) => {
677                    // Replay error to create counterexample and to collect logs, traces and
678                    // coverage.
679                    match replay_error(
680                        &case_data,
681                        &invariant_contract,
682                        self.clone_executor(),
683                        &self.cr.mcr.known_contracts,
684                        identified_contracts.clone(),
685                        &mut self.result.logs,
686                        &mut self.result.traces,
687                        &mut self.result.coverage,
688                        &mut self.result.deprecated_cheatcodes,
689                        progress.as_ref(),
690                        show_solidity,
691                    ) {
692                        Ok(call_sequence) => {
693                            if !call_sequence.is_empty() {
694                                // Persist error in invariant failure dir.
695                                if let Err(err) = foundry_common::fs::create_dir_all(failure_dir) {
696                                    error!(%err, "Failed to create invariant failure dir");
697                                } else if let Err(err) = foundry_common::fs::write_json_file(
698                                    failure_file.as_path(),
699                                    &call_sequence,
700                                ) {
701                                    error!(%err, "Failed to record call sequence");
702                                }
703
704                                let original_seq_len =
705                                    if let TestError::Fail(_, calls) = &case_data.test_error {
706                                        calls.len()
707                                    } else {
708                                        call_sequence.len()
709                                    };
710
711                                counterexample =
712                                    Some(CounterExample::Sequence(original_seq_len, call_sequence))
713                            }
714                        }
715                        Err(err) => {
716                            error!(%err, "Failed to replay invariant error");
717                        }
718                    };
719                }
720                InvariantFuzzError::MaxAssumeRejects(_) => {}
721            },
722
723            // If invariants ran successfully, replay the last run to collect logs and
724            // traces.
725            _ => {
726                if let Err(err) = replay_run(
727                    &invariant_contract,
728                    self.clone_executor(),
729                    &self.cr.mcr.known_contracts,
730                    identified_contracts.clone(),
731                    &mut self.result.logs,
732                    &mut self.result.traces,
733                    &mut self.result.coverage,
734                    &mut self.result.deprecated_cheatcodes,
735                    &invariant_result.last_run_inputs,
736                    show_solidity,
737                ) {
738                    error!(%err, "Failed to replay last invariant run");
739                }
740            }
741        }
742
743        self.result.invariant_result(
744            invariant_result.gas_report_traces,
745            success,
746            reason,
747            counterexample,
748            invariant_result.cases,
749            invariant_result.reverts,
750            invariant_result.metrics,
751        );
752        self.result
753    }
754
755    /// Runs a fuzzed test.
756    ///
757    /// Applies the before test txes (if any), fuzzes the current function and returns the
758    /// `TestResult`.
759    ///
760    /// Before test txes are applied in order and state modifications committed to the EVM database
761    /// (therefore the fuzz test will use the modified state).
762    /// State modifications of before test txes and fuzz test are discarded after test ends,
763    /// similar to `eth_call`.
764    fn run_fuzz_test(mut self, func: &Function) -> TestResult {
765        // Prepare fuzz test execution.
766        if self.prepare_test(func).is_err() {
767            return self.result;
768        }
769
770        let runner = self.fuzz_runner();
771        let fuzz_config = self.config.fuzz.clone();
772
773        let progress =
774            start_fuzz_progress(self.cr.progress, self.cr.name, &func.name, fuzz_config.runs);
775
776        // Run fuzz test.
777        let fuzzed_executor =
778            FuzzedExecutor::new(self.executor.into_owned(), runner, self.tcfg.sender, fuzz_config);
779        let result = fuzzed_executor.fuzz(
780            func,
781            &self.setup.fuzz_fixtures,
782            &self.setup.deployed_libs,
783            self.address,
784            &self.cr.mcr.revert_decoder,
785            progress.as_ref(),
786        );
787        self.result.fuzz_result(result);
788        self.result
789    }
790
791    /// Prepares single unit test and fuzz test execution:
792    /// - set up the test result and executor
793    /// - check if before test txes are configured and apply them in order
794    ///
795    /// Before test txes are arrays of arbitrary calldata obtained by calling the `beforeTest`
796    /// function with test selector as a parameter.
797    ///
798    /// Unit tests within same contract (or even current test) are valid options for before test tx
799    /// configuration. Test execution stops if any of before test txes fails.
800    fn prepare_test(&mut self, func: &Function) -> Result<(), ()> {
801        let address = self.setup.address;
802
803        // Apply before test configured functions (if any).
804        if self.cr.contract.abi.functions().filter(|func| func.name.is_before_test_setup()).count() ==
805            1
806        {
807            for calldata in self
808                .executor
809                .call_sol_default(
810                    address,
811                    &ITest::beforeTestSetupCall { testSelector: func.selector() },
812                )
813                .beforeTestCalldata
814            {
815                // Apply before test configured calldata.
816                match self.executor.to_mut().transact_raw(
817                    self.tcfg.sender,
818                    address,
819                    calldata,
820                    U256::ZERO,
821                ) {
822                    Ok(call_result) => {
823                        let reverted = call_result.reverted;
824
825                        // Merge tx result traces in unit test result.
826                        self.result.extend(call_result);
827
828                        // To continue unit test execution the call should not revert.
829                        if reverted {
830                            self.result.single_fail(None);
831                            return Err(());
832                        }
833                    }
834                    Err(_) => {
835                        self.result.single_fail(None);
836                        return Err(());
837                    }
838                }
839            }
840        }
841        Ok(())
842    }
843
844    fn fuzz_runner(&self) -> TestRunner {
845        let config = &self.config.fuzz;
846        let failure_persist_path = config
847            .failure_persist_dir
848            .as_ref()
849            .unwrap()
850            .join(config.failure_persist_file.as_ref().unwrap())
851            .into_os_string()
852            .into_string()
853            .unwrap();
854        fuzzer_with_cases(
855            config.seed,
856            config.runs,
857            config.max_test_rejects,
858            Some(Box::new(FileFailurePersistence::Direct(failure_persist_path.leak()))),
859        )
860    }
861
862    fn invariant_runner(&self) -> TestRunner {
863        let config = &self.config.invariant;
864        fuzzer_with_cases(self.config.fuzz.seed, config.runs, config.max_assume_rejects, None)
865    }
866
867    fn clone_executor(&self) -> Executor {
868        self.executor.clone().into_owned()
869    }
870}
871
872fn fuzzer_with_cases(
873    seed: Option<U256>,
874    cases: u32,
875    max_global_rejects: u32,
876    file_failure_persistence: Option<Box<dyn FailurePersistence>>,
877) -> TestRunner {
878    let config = proptest::test_runner::Config {
879        failure_persistence: file_failure_persistence,
880        cases,
881        max_global_rejects,
882        // Disable proptest shrink: for fuzz tests we provide single counterexample,
883        // for invariant tests we shrink outside proptest.
884        max_shrink_iters: 0,
885        ..Default::default()
886    };
887
888    if let Some(seed) = seed {
889        trace!(target: "forge::test", %seed, "building deterministic fuzzer");
890        let rng = TestRng::from_seed(RngAlgorithm::ChaCha, &seed.to_be_bytes::<32>());
891        TestRunner::new_with_rng(config, rng)
892    } else {
893        trace!(target: "forge::test", "building stochastic fuzzer");
894        TestRunner::new(config)
895    }
896}
